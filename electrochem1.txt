<script>
function startGalvanic() {

  // Zn2+ moves away from Zn electrode (oxidation)
  document.getElementById("zn1").style.left = "120px";
  document.getElementById("zn2").style.left = "150px";

  // Cu2+ moves toward Cu electrode (reduction)
  document.getElementById("cu1").style.left = "240px";
  document.getElementById("cu2").style.left = "210px";

  // Electron flow in wire (Zn → Cu)
  let e = document.getElementById("wireElectron");
  e.style.transition = "left 2s linear";
  e.style.left = "90%";
}
</script>
